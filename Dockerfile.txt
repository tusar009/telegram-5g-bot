# Use a lightweight Python image
FROM python:3.10-slim

# Install system dependencies (Chromium)
RUN apt update && apt install -y chromium-browser && rm -rf /var/lib/apt/lists/*

# Set Chromium path environment variable
ENV GOOGLE_CHROME_BIN="/usr/bin/chromium-browser"

# Set working directory
WORKDIR /app

# Copy all bot files
COPY . /app

# Install required Python packages
RUN pip install --no-cache-dir -r requirements.txt

# Health check (ensures bot is running)
HEALTHCHECK CMD pgrep -f "python map_generator.py" || exit 1

# Run the bot
CMD ["python", "map_generator.py"]
